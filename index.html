Setup Marketplace — SQL Supabase

Execute o SQL abaixo no SQL Editor do Supabase para criar as tabelas do Marketplace: 

-- =============================================
-- LEGACY MARKETPLACE — SETUP COMPLETO SUPABASE
-- Execute no SQL Editor do Supabase
-- =============================================

-- 1. Tabela de Vendedores
CREATE TABLE IF NOT EXISTS public.marketplace_vendedores (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    nome TEXT NOT NULL,
    email TEXT NOT NULL,
    telefone TEXT,
    status TEXT DEFAULT 'ativo' CHECK (status IN ('ativo','inativo')),
    data_criacao TIMESTAMPTZ DEFAULT now()
);

-- 2. Tabela de Produtos
CREATE TABLE IF NOT EXISTS public.marketplace_produtos (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    nome TEXT NOT NULL,
    descricao TEXT,
    preco NUMERIC(10,2) DEFAULT 0 CHECK (preco >= 0),
    imagem_url TEXT,
    imagem2_url TEXT,
    imagem3_url TEXT,
    imagem4_url TEXT,
    imagem5_url TEXT,
    categoria TEXT,
    vendedor_nome TEXT,
    vendedor_id UUID REFERENCES public.marketplace_vendedores(id) ON DELETE SET NULL,
    status TEXT DEFAULT 'ativo' CHECK (status IN ('ativo','inativo')),
    data_criacao TIMESTAMPTZ DEFAULT now()
);

-- Migração: adiciona colunas de imagem extra se ainda não existirem (banco já criado)
ALTER TABLE public.marketplace_produtos ADD COLUMN IF NOT EXISTS imagem2_url TEXT;
ALTER TABLE public.marketplace_produtos ADD COLUMN IF NOT EXISTS imagem3_url TEXT;
ALTER TABLE public.marketplace_produtos ADD COLUMN IF NOT EXISTS imagem4_url TEXT;
ALTER TABLE public.marketplace_produtos ADD COLUMN IF NOT EXISTS imagem5_url TEXT;
ALTER TABLE public.marketplace_produtos ADD COLUMN IF NOT EXISTS whatsapp TEXT;
ALTER TABLE public.marketplace_produtos ADD COLUMN IF NOT EXISTS pix_key TEXT;
ALTER TABLE public.marketplace_produtos ADD COLUMN IF NOT EXISTS incluso TEXT;
ALTER TABLE public.marketplace_produtos ADD COLUMN IF NOT EXISTS extras TEXT;

-- Índices para performance
CREATE INDEX IF NOT EXISTS idx_mkp_produtos_status ON public.marketplace_produtos(status);
CREATE INDEX IF NOT EXISTS idx_mkp_produtos_categoria ON public.marketplace_produtos(categoria);
CREATE INDEX IF NOT EXISTS idx_mkp_produtos_vendedor ON public.marketplace_produtos(vendedor_id);

-- 3. Habilitar RLS
ALTER TABLE public.marketplace_vendedores ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.marketplace_produtos ENABLE ROW LEVEL SECURITY;

-- 4. Políticas RLS — Leitura pública (anon)
DROP POLICY IF EXISTS "mkp_vendedores_select_public" ON public.marketplace_vendedores;
CREATE POLICY "mkp_vendedores_select_public" ON public.marketplace_vendedores
    FOR SELECT TO anon, authenticated USING (true);

DROP POLICY IF EXISTS "mkp_produtos_select_public" ON public.marketplace_produtos;
CREATE POLICY "mkp_produtos_select_public" ON public.marketplace_produtos
    FOR SELECT TO anon, authenticated USING (status = 'ativo');

-- 5. Políticas RLS — CRUD completo para usuários autenticados (admin)
DROP POLICY IF EXISTS "mkp_vendedores_admin_all" ON public.marketplace_vendedores;
CREATE POLICY "mkp_vendedores_admin_all" ON public.marketplace_vendedores
    FOR ALL TO authenticated USING (true) WITH CHECK (true);

DROP POLICY IF EXISTS "mkp_produtos_admin_all" ON public.marketplace_produtos;
CREATE POLICY "mkp_produtos_admin_all" ON public.marketplace_produtos
    FOR ALL TO authenticated USING (true) WITH CHECK (true);

-- 6. Storage Bucket para imagens dos produtos
-- (Execute este bloco separadamente se o anterior der erro)
INSERT INTO storage.buckets (id, name, public, file_size_limit, allowed_mime_types)
VALUES (
    'marketplace-fotos',
    'marketplace-fotos',
    true,
    5242880,
    ARRAY['image/jpeg','image/png','image/webp','image/gif','image/avif']
) ON CONFLICT (id) DO UPDATE SET
    public = true,
    file_size_limit = 5242880;

-- 7. Políticas Storage — leitura pública
DROP POLICY IF EXISTS "mkp_fotos_public_read" ON storage.objects;
CREATE POLICY "mkp_fotos_public_read" ON storage.objects
    FOR SELECT TO anon, authenticated
    USING (bucket_id = 'marketplace-fotos');

-- 8. Políticas Storage — upload/delete apenas autenticados
DROP POLICY IF EXISTS "mkp_fotos_auth_insert" ON storage.objects;
CREATE POLICY "mkp_fotos_auth_insert" ON storage.objects
    FOR INSERT TO authenticated
    WITH CHECK (bucket_id = 'marketplace-fotos');

DROP POLICY IF EXISTS "mkp_fotos_auth_update" ON storage.objects;
CREATE POLICY "mkp_fotos_auth_update" ON storage.objects
    FOR UPDATE TO authenticated
    USING (bucket_id = 'marketplace-fotos');

DROP POLICY IF EXISTS "mkp_fotos_auth_delete" ON storage.objects;
CREATE POLICY "mkp_fotos_auth_delete" ON storage.objects
    FOR DELETE TO authenticated
    USING (bucket_id = 'marketplace-fotos');

-- =============================================
-- SETUP CONCLUÍDO ✓
-- Tabelas: public.marketplace_produtos
--          public.marketplace_vendedores
-- Bucket:  marketplace-fotos (público)
-- =============================================
