Setup SQL — Tabela questionarios_ia

Execute no SQL Editor do Supabase para criar a tabela do módulo IA PROJECT:

-- =============================================
-- IA PROJECT — Tabela questionarios_ia
-- Execute no SQL Editor do Supabase
-- =============================================

CREATE TABLE IF NOT EXISTS public.questionarios_ia (
  id             uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  token          text NOT NULL UNIQUE,
  nome_cliente   text,
  respostas      jsonb,
  prompt_gerado  text,
  imagem_url     text,
  status         text DEFAULT 'pendente' CHECK (status IN ('pendente','respondido','visualizado')),
  expires_at     timestamptz,
  created_at     timestamptz DEFAULT now()
);

-- Índices
CREATE INDEX IF NOT EXISTS idx_questionarios_ia_token  ON public.questionarios_ia(token);
CREATE INDEX IF NOT EXISTS idx_questionarios_ia_status ON public.questionarios_ia(status);

-- RLS (Row Level Security)
ALTER TABLE public.questionarios_ia ENABLE ROW LEVEL SECURITY;

-- Policy: leitura e escrita pública (via token) — ajuste conforme sua política
CREATE POLICY "Acesso público via token" ON public.questionarios_ia
  FOR ALL USING (true) WITH CHECK (true);

-- =============================================
-- SETUP CONCLUÍDO ✓ — Tabela: questionarios_ia
-- =============================================
