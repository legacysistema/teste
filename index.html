Execute este SQL no Supabase para criar as tabelas:

-- Tabelas FotoSelect CREATE TABLE IF NOT EXISTS fs_clients ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, name TEXT NOT NULL, event TEXT, email TEXT, event_date DATE, included_count INTEGER DEFAULT 20, price_per_extra DECIMAL(10,2) DEFAULT 25.00, created_at TIMESTAMP DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS fs_photos ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, client_id UUID REFERENCES fs_clients(id) ON DELETE CASCADE, filename TEXT, storage_url TEXT, photo_order INTEGER, created_at TIMESTAMP DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS fs_selections ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, client_id UUID REFERENCES fs_clients(id) ON DELETE CASCADE UNIQUE, selected_photo_ids TEXT[], extras_count INTEGER DEFAULT 0, total_value DECIMAL(10,2) DEFAULT 0, payment_status TEXT DEFAULT 'pending', confirmed_at TIMESTAMP, created_at TIMESTAMP DEFAULT NOW() ); INSERT INTO storage.buckets (id, name, public) VALUES ('galeria', 'galeria', true) ON CONFLICT DO NOTHING; ALTER TABLE fs_clients ENABLE ROW LEVEL SECURITY; ALTER TABLE fs_photos ENABLE ROW LEVEL SECURITY; ALTER TABLE fs_selections ENABLE ROW LEVEL SECURITY; CREATE POLICY "fs_clients_all" ON fs_clients FOR ALL TO anon USING (true) WITH CHECK (true); CREATE POLICY "fs_photos_all" ON fs_photos FOR ALL TO anon USING (true) WITH CHECK (true); CREATE POLICY "fs_selections_all" ON fs_selections FOR ALL TO anon USING (true) WITH CHECK (true); CREATE POLICY "fs_storage_insert" ON storage.objects FOR INSERT TO anon WITH CHECK (bucket_id = 'galeria'); CREATE POLICY "fs_storage_select" ON storage.objects FOR SELECT TO anon USING (bucket_id = 'galeria'); CREATE POLICY "fs_storage_delete" ON storage.objects FOR DELETE TO anon USING (bucket_id = 'galeria');
